-- Horizon Systems - Final Schema (Base64 Optimized)
-- Duration Tiers: 6 Months, 1 Year, 3 Years

SET SQL_MODE = "NO_AUTO_VALUE_ON_ZERO";
START TRANSACTION;
SET time_zone = "+00:00";

-- --------------------------------------------------------

CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(150) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    first_name VARCHAR(100) NOT NULL,
    middle_name VARCHAR(100),
    last_name VARCHAR(100) NOT NULL,
    contact_number VARCHAR(20) NOT NULL,
    profile_picture LONGTEXT, -- Changed to LONGTEXT for Base64
    is_verified TINYINT(1) NOT NULL DEFAULT 0,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL
);

CREATE TABLE roles (
    role_id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(50) NOT NULL
);

CREATE TABLE gym_addresses (
    address_id INT AUTO_INCREMENT PRIMARY KEY,
    address_line VARCHAR(255) NOT NULL,
    barangay VARCHAR(100) NOT NULL,
    city VARCHAR(100) NOT NULL,
    province VARCHAR(100) NOT NULL,
    region VARCHAR(100) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL
);

CREATE TABLE website_plans (
    website_plan_id INT AUTO_INCREMENT PRIMARY KEY,
    plan_name VARCHAR(100) NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    billing_cycle VARCHAR(100) NOT NULL,
    duration_months INT NOT NULL,
    features TEXT NOT NULL,
    is_active TINYINT(1) NOT NULL DEFAULT 1
);

CREATE TABLE membership_plan_types (
    plan_type_id INT AUTO_INCREMENT PRIMARY KEY,
    type_name VARCHAR(50) NOT NULL
);

CREATE TABLE service_catalog (
    catalog_service_id INT AUTO_INCREMENT PRIMARY KEY,
    service_name VARCHAR(100) NOT NULL,
    service_category VARCHAR(50) NOT NULL,
    default_description TEXT NOT NULL
);

CREATE TABLE gym_owner_applications (
    application_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_name VARCHAR(100) NOT NULL,
    business_name VARCHAR(100) NOT NULL,
    business_type VARCHAR(50) NOT NULL,
    address_id INT NOT NULL,
    owner_valid_id_type VARCHAR(50) NOT NULL,
    bir_number VARCHAR(50) NOT NULL,
    business_permit_no VARCHAR(100) NOT NULL,
    contact_number VARCHAR(20) NOT NULL,
    email VARCHAR(150) NOT NULL,
    application_status VARCHAR(50) NOT NULL,
    submitted_at DATETIME NOT NULL,
    reviewed_by INT,
    reviewed_at DATETIME,
    remarks TEXT,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (address_id) REFERENCES gym_addresses(address_id),
    FOREIGN KEY (reviewed_by) REFERENCES users(user_id)
);

CREATE TABLE application_documents (
    document_id INT AUTO_INCREMENT PRIMARY KEY,
    application_id INT NOT NULL,
    document_type VARCHAR(50) NOT NULL,
    file_path LONGTEXT NOT NULL, 
    uploaded_at DATETIME NOT NULL,
    FOREIGN KEY (application_id) REFERENCES gym_owner_applications(application_id)
);

CREATE TABLE gyms (
    gym_id INT AUTO_INCREMENT PRIMARY KEY,
    owner_user_id INT NOT NULL,
    application_id INT,
    gym_name VARCHAR(100) NOT NULL,
    business_name VARCHAR(100) NOT NULL,
    description TEXT,
    address_id INT NOT NULL,
    contact_number VARCHAR(20) NOT NULL,
    email VARCHAR(150) NOT NULL,
    profile_picture LONGTEXT, 
    tenant_code VARCHAR(50) NOT NULL,
    status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (owner_user_id) REFERENCES users(user_id),
    FOREIGN KEY (application_id) REFERENCES gym_owner_applications(application_id),
    FOREIGN KEY (address_id) REFERENCES gym_addresses(address_id)
);

CREATE TABLE user_roles (
    user_role_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    role_id INT NOT NULL,
    gym_id INT,
    role_status VARCHAR(50) NOT NULL,
    assigned_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (role_id) REFERENCES roles(role_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE gym_details (
    gym_detail_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    opening_time TIME NOT NULL,
    closing_time TIME NOT NULL,
    max_capacity INT NOT NULL,
    has_lockers TINYINT(1) NOT NULL DEFAULT 0,
    has_shower TINYINT(1) NOT NULL DEFAULT 0,
    has_parking TINYINT(1) NOT NULL DEFAULT 0,
    has_wifi TINYINT(1) NOT NULL DEFAULT 0,
    about_text TEXT,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE gym_rules (
    rule_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    allow_member_booking TINYINT(1) NOT NULL DEFAULT 1,
    allow_walk_in TINYINT(1) NOT NULL DEFAULT 1,
    allow_coach_application TINYINT(1) NOT NULL DEFAULT 1,
    show_coach_schedule_to_members TINYINT(1) NOT NULL DEFAULT 1,
    membership_required_for_booking TINYINT(1) NOT NULL DEFAULT 0,
    allow_part_time_coach TINYINT(1) NOT NULL DEFAULT 1,
    allow_professional_coach TINYINT(1) NOT NULL DEFAULT 1,
    max_booking_per_day INT NOT NULL,
    cancellation_hours_limit INT NOT NULL,
    late_minutes_limit INT NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE client_subscriptions (
    client_subscription_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    owner_user_id INT NOT NULL,
    website_plan_id INT NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE,
    subscription_status VARCHAR(50) NOT NULL,
    payment_status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (owner_user_id) REFERENCES users(user_id),
    FOREIGN KEY (website_plan_id) REFERENCES website_plans(website_plan_id)
);

CREATE TABLE tenant_pages (
    page_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    page_slug VARCHAR(50) NOT NULL,
    page_title VARCHAR(100) NOT NULL,
    logo_path LONGTEXT, 
    banner_image LONGTEXT, 
    theme_color VARCHAR(100) NOT NULL,
    secondary_color VARCHAR(20) DEFAULT '#14121a',
    font_family VARCHAR(50) DEFAULT 'Lexend',
    border_radius VARCHAR(20) DEFAULT '24px',
    theme_mode ENUM('dark','light') DEFAULT 'dark',
    home_title TEXT DEFAULT NULL,
    home_subtitle TEXT DEFAULT NULL,
    about_text TEXT,
    contact_text TEXT,
    footer_text TEXT DEFAULT NULL,
    portal_tab_text VARCHAR(50) DEFAULT 'Gym Portal',
    app_download_link VARCHAR(255),
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE gym_services (
    gym_service_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    catalog_service_id INT NOT NULL,
    custom_service_name VARCHAR(100),
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    duration_minutes INT NOT NULL,
    max_participants INT NOT NULL,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (catalog_service_id) REFERENCES service_catalog(catalog_service_id)
);

CREATE TABLE membership_plans (
    membership_plan_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    plan_name VARCHAR(100) NOT NULL,
    plan_type_id INT NOT NULL,
    duration_value INT NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    session_limit INT,
    description TEXT,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (plan_type_id) REFERENCES membership_plan_types(plan_type_id)
);

CREATE TABLE staff_invitations (
    invitation_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    email VARCHAR(150) NOT NULL,
    staff_role VARCHAR(50) NOT NULL,
    invited_by INT NOT NULL,
    token VARCHAR(50) NOT NULL,
    invitation_status VARCHAR(50) NOT NULL,
    expires_at DATETIME NOT NULL,
    accepted_at DATETIME,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (invited_by) REFERENCES users(user_id)
);

CREATE TABLE staff (
    staff_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_id INT NOT NULL,
    staff_role VARCHAR(50) NOT NULL,
    employment_type VARCHAR(50) NOT NULL,
    hire_date DATE NOT NULL,
    end_date DATE,
    status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE user_verifications (
    verification_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_id INT,
    verification_type VARCHAR(50) NOT NULL,
    code VARCHAR(50) NOT NULL,
    status VARCHAR(50) NOT NULL,
    expires_at DATETIME NOT NULL,
    verified_at DATETIME,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE member_registrations (
    member_registration_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    user_id INT NOT NULL,
    email VARCHAR(150) NOT NULL,
    registration_source VARCHAR(100) NOT NULL,
    registered_by_user_id INT,
    verification_id INT,
    registration_status VARCHAR(50) NOT NULL,
    completed_at DATETIME,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (registered_by_user_id) REFERENCES users(user_id),
    FOREIGN KEY (verification_id) REFERENCES user_verifications(verification_id)
);

CREATE TABLE members (
    member_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_id INT NOT NULL,
    member_code VARCHAR(50) NOT NULL,
    birth_date DATE NOT NULL,
    sex VARCHAR(100) NOT NULL,
    occupation VARCHAR(100),
    address VARCHAR(255),
    emergency_contact_name VARCHAR(100) NOT NULL,
    emergency_contact_number VARCHAR(20) NOT NULL,
    medical_history TEXT,
    member_status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE member_subscriptions (
    subscription_id INT AUTO_INCREMENT PRIMARY KEY,
    member_id INT NOT NULL,
    membership_plan_id INT NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE,
    sessions_total INT NOT NULL,
    sessions_used INT NOT NULL DEFAULT 0,
    subscription_status VARCHAR(50) NOT NULL,
    payment_status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (member_id) REFERENCES members(member_id),
    FOREIGN KEY (membership_plan_id) REFERENCES membership_plans(membership_plan_id)
);

CREATE TABLE coach_applications (
    coach_application_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_id INT NOT NULL,
    coach_type VARCHAR(50) NOT NULL,
    specialization VARCHAR(100) NOT NULL,
    license_number VARCHAR(50),
    certification_file LONGTEXT NOT NULL, 
    valid_until DATE,
    application_status VARCHAR(50) NOT NULL,
    submitted_at DATETIME NOT NULL,
    reviewed_by INT,
    reviewed_at DATETIME,
    remarks TEXT,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (reviewed_by) REFERENCES users(user_id)
);

CREATE TABLE coaches (
    coach_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_id INT NOT NULL,
    coach_application_id INT,
    coach_type VARCHAR(50) NOT NULL,
    specialization VARCHAR(100) NOT NULL,
    license_number VARCHAR(50),
    valid_until DATE,
    hire_date DATE NOT NULL,
    end_date DATE,
    status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (coach_application_id) REFERENCES coach_applications(coach_application_id)
);

CREATE TABLE coach_schedules (
    coach_schedule_id INT AUTO_INCREMENT PRIMARY KEY,
    coach_id INT NOT NULL,
    day_of_week VARCHAR(100) NOT NULL,
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    availability_status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (coach_id) REFERENCES coaches(coach_id)
);

CREATE TABLE class_schedules (
    class_schedule_id INT AUTO_INCREMENT PRIMARY KEY,
    gym_id INT NOT NULL,
    gym_service_id INT NOT NULL,
    coach_id INT,
    class_title VARCHAR(100) NOT NULL,
    schedule_date DATE NOT NULL,
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    max_capacity INT NOT NULL,
    status VARCHAR(50) NOT NULL,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (gym_service_id) REFERENCES gym_services(gym_service_id),
    FOREIGN KEY (coach_id) REFERENCES coaches(coach_id)
);

CREATE TABLE bookings (
    booking_id INT AUTO_INCREMENT PRIMARY KEY,
    member_id INT NOT NULL,
    gym_id INT NOT NULL,
    gym_service_id INT NOT NULL,
    class_schedule_id INT,
    coach_id INT,
    booking_reference VARCHAR(50) NOT NULL,
    booking_date DATE NOT NULL,
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    booking_source VARCHAR(100) NOT NULL,
    booking_status VARCHAR(50) NOT NULL,
    cancellation_reason VARCHAR(100),
    approved_by INT,
    approved_at DATETIME,
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (member_id) REFERENCES members(member_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (gym_service_id) REFERENCES gym_services(gym_service_id),
    FOREIGN KEY (class_schedule_id) REFERENCES class_schedules(class_schedule_id),
    FOREIGN KEY (coach_id) REFERENCES coaches(coach_id),
    FOREIGN KEY (approved_by) REFERENCES users(user_id)
);

CREATE TABLE attendance (
    attendance_id INT AUTO_INCREMENT PRIMARY KEY,
    member_id INT NOT NULL,
    gym_id INT NOT NULL,
    booking_id INT,
    attendance_date DATE NOT NULL,
    check_in_time TIME NOT NULL,
    check_out_time TIME,
    attendance_status VARCHAR(50) NOT NULL,
    recorded_by INT,
    remarks TEXT,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (member_id) REFERENCES members(member_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (booking_id) REFERENCES bookings(booking_id),
    FOREIGN KEY (recorded_by) REFERENCES users(user_id)
);

CREATE TABLE payments (
    payment_id INT AUTO_INCREMENT PRIMARY KEY,
    member_id INT,
    gym_id INT,
    subscription_id INT,
    booking_id INT,
    client_subscription_id INT,
    amount DECIMAL(10,2) NOT NULL,
    payment_method VARCHAR(50) NOT NULL,
    payment_type VARCHAR(50) NOT NULL,
    reference_number VARCHAR(50) NOT NULL,
    receipt_image LONGTEXT, 
    payment_status VARCHAR(50) NOT NULL,
    verified_by INT,
    verified_at DATETIME,
    remarks TEXT,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (member_id) REFERENCES members(member_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (subscription_id) REFERENCES member_subscriptions(subscription_id),
    FOREIGN KEY (booking_id) REFERENCES bookings(booking_id),
    FOREIGN KEY (client_subscription_id) REFERENCES client_subscriptions(client_subscription_id),
    FOREIGN KEY (verified_by) REFERENCES users(user_id)
);

CREATE TABLE notifications (
    notification_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    gym_id INT,
    title VARCHAR(100) NOT NULL,
    message TEXT NOT NULL,
    notification_type VARCHAR(50) NOT NULL,
    is_read TINYINT(1) NOT NULL DEFAULT 0,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE audit_logs (
    audit_log_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    gym_id INT,
    action_type VARCHAR(50) NOT NULL,
    table_name VARCHAR(100) NOT NULL,
    record_id INT NOT NULL,
    old_values LONGTEXT NOT NULL,
    new_values LONGTEXT NOT NULL,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id)
);

CREATE TABLE feedback (
    feedback_id INT AUTO_INCREMENT PRIMARY KEY,
    member_id INT NOT NULL,
    gym_id INT NOT NULL,
    coach_id INT,
    rating DECIMAL(3,2) NOT NULL,
    category VARCHAR(50) NOT NULL,
    comment TEXT NOT NULL,
    is_anonymous TINYINT(1) NOT NULL DEFAULT 0,
    is_visible TINYINT(1) NOT NULL DEFAULT 1,
    admin_reply TEXT,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (member_id) REFERENCES members(member_id),
    FOREIGN KEY (gym_id) REFERENCES gyms(gym_id),
    FOREIGN KEY (coach_id) REFERENCES coaches(coach_id)
);
